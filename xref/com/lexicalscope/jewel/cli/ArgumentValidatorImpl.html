15  package com.lexicalscope.jewel.cli;
16  
17  import java.util.ArrayList;
18  import java.util.Iterator;
19  import java.util.LinkedHashMap;
20  import java.util.List;
21  import java.util.Map;
22  
23  class ArgumentValidatorImpl<O> implements ArgumentValidator<O>
24  {
25      private final ValidationErrorBuilder m_validationErrorBuilder;
26      private final List<String> m_validatedUnparsedArguments;
27      private final Map<String, List<String>> m_validatedArguments;
28      private final OptionsSpecification<O> m_specification;
29  
30      public ArgumentValidatorImpl(final OptionsSpecification<O> specification)
31      {
32          m_specification = specification;
33  
34          m_validatedArguments = new LinkedHashMap<String, List<String>>();
35          m_validatedUnparsedArguments = new ArrayList<String>();
36  
37          m_validationErrorBuilder = new ValidationErrorBuilderImpl();
38      }
39  
40      /**
41       * {@inheritDoc}
42       */
43      public ArgumentCollection validateArguments(final ArgumentCollection arguments) throws ArgumentValidationException
44      {
45          m_validatedUnparsedArguments.addAll(arguments.getUnparsed());
46  
47          final Iterator<Argument> argumentsIterator = arguments.iterator();
48          while (argumentsIterator.hasNext())
49          {
50              final Argument argument = argumentsIterator.next();
51              final boolean isLast = !argumentsIterator.hasNext();
52  
53              if (!m_specification.isSpecified(argument.getOptionName()))
54              {
55                  m_validationErrorBuilder.unexpectedOption(argument.getOptionName());
56              }
57              else
58              {
59                  final ParsedOptionSpecification optionSpecification =
60                          m_specification.getSpecification(argument.getOptionName());
61                  if (optionSpecification.isHelpOption())
62                  {
63                      throw new HelpRequestedException(m_specification);
64                  }
65                  else if (argument.getValues().size() == 0
66                          && optionSpecification.hasValue()
67                          && !optionSpecification.isMultiValued())
68                  {
69                      m_validationErrorBuilder.missingValue(optionSpecification);
70                  }
71                  else if (!isLast && argument.getValues().size() > 0 && !optionSpecification.hasValue())
72                  {
73                      m_validationErrorBuilder.unexpectedValue(optionSpecification, argument.getValues());
74                  }
75                  else if (!isLast && argument.getValues().size() > 1 && !optionSpecification.isMultiValued())
76                  {
77                      m_validationErrorBuilder.unexpectedAdditionalValues(optionSpecification, argument.getValues().subList(1, argument.getValues().size()));
78                  }
79  
80                  if (isLast && hasExcessValues(argument, optionSpecification))
81                  {
82                      final List<String> values = new ArrayList<String>();
83                      final List<String> unparsed;
84                      if (optionSpecification.hasValue())
85                      {
86                          values.add(argument.getValues().get(0));
87                          unparsed = new ArrayList<String>(argument.getValues().subList(1, argument.getValues().size()));
88                      }
89                      else
90                      {
91                          unparsed = argument.getValues();
92                      }
93  
94                      m_validatedArguments.put(argument.getOptionName(), values);
95                      m_validatedUnparsedArguments.addAll(0, unparsed);
96                  }
97                  else
98                  {
99                      checkAndAddValues(
100                             optionSpecification,
101                             argument.getOptionName(),
102                             new ArrayList<String>(argument.getValues()));
103                 }
104             }
105         }
106 
107         for (final ParsedOptionSpecification optionSpecification : m_specification.getMandatoryOptions())
108         {
109             if (!arguments.containsAny(optionSpecification.getNames()))
110             {
111                 m_validationErrorBuilder.missingOption(optionSpecification);
112             }
113         }
114 
115         validateUnparsedOptions();
116 
117         m_validationErrorBuilder.validate();
118 
119         return new ArgumentCollectionImpl(m_validatedArguments, m_validatedUnparsedArguments);
120     }
121 
122     private void validateUnparsedOptions()
123     {
124         if (m_specification.hasUnparsedSpecification())
125         {
126             final OptionSpecification argumentSpecification = m_specification.getUnparsedSpecification();
127 
128             if (!argumentSpecification.isMultiValued()
129                     && !argumentSpecification.isOptional()
130                     && m_validatedUnparsedArguments.isEmpty())
131             {
132                 m_validationErrorBuilder.missingValue(argumentSpecification);
133             }
134             else if (!argumentSpecification.isMultiValued() && m_validatedUnparsedArguments.size() > 1)
135             {
136                 m_validationErrorBuilder.unexpectedAdditionalValues(argumentSpecification, m_validatedUnparsedArguments.subList(1, m_validatedUnparsedArguments.size()));
137             }
138         }
139         else if (!m_validatedUnparsedArguments.isEmpty())
140         {
141             m_validationErrorBuilder.unexpectedTrailingValue(m_validatedUnparsedArguments);
142         }
143     }
144 
145     private boolean hasExcessValues(final Argument entry, final ParsedOptionSpecification optionSpecification)
146     {
147         return !optionSpecification.isMultiValued()
148                 && (entry.getValues().size() > 1 || entry.getValues().size() > 0 && !optionSpecification.hasValue());
149     }
150 
151     private void checkAndAddValues(
152             final ParsedOptionSpecification optionSpecification,
153             final String option,
154             final ArrayList<String> values)
155     {
156         for (final String value : values)
157         {
158             if (!patternMatches(optionSpecification, value))
159             {
160                 m_validationErrorBuilder.patternMismatch(optionSpecification, value);
161             }
162         }
163         m_validatedArguments.put(option, new ArrayList<String>(values));
164     }
165 
166     private boolean patternMatches(final ParsedOptionSpecification optionSpecification, final String value)
167     {
168         return value.matches(optionSpecification.getPattern());
169     }
