27  class OptionAnnotationAdapter extends AbstractOptionAdapter {
28      private final Option option;
29  
30      OptionAnnotationAdapter(
31              final FluentClass<?> klass,
32              final FluentMethod method) {
33          super(klass, method);
34          this.option = method.annotation(Option.class);
35      }
36  
37      @Override public String description() {
38          return option.description().trim();
39      }
40  
41      @Override public String pattern() {
42          return option.pattern();
43      }
44  
45      @Override public boolean defaultToNull() {
46          return option.defaultToNull();
47      }
48  
49      @Override public String[] defaultValue() {
50          return option.defaultValue();
51      }
52  
53      public List<String> shortName() {
54          final List<String> shortNames = new ArrayList<String>();
55          for (final String element : option.shortName()) {
56              final String shortName = element.trim();
57              if (shortName.length() > 0) {
58                  shortNames.add(element.substring(0, 1));
59              }
60          }
61          return shortNames;
62      }
63  
64      public List<String> longName() {
65          return option.longName().length == 0
66                  ? asList(method.property())
67                  : asList(option.longName());
68      }
69  
70      public boolean helpRequest() {
71          return option.helpRequest();
72      }
73  
74      @Override public boolean isHidden() {
75          return option.hidden();
76      }
77  
78      @Override public int minimum() {
79          return option.minimum();
80      }
81  
82      @Override public int exactly() {
83          return option.exactly();
84      }
85  
86      @Override public int maximum() {
87          return option.maximum();
88      }
89  }
