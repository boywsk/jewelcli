15  package uk.co.flamingpenguin.jewel.cli;
16  
17  import java.util.ArrayList;
18  import java.util.Iterator;
19  import java.util.LinkedHashMap;
20  import java.util.List;
21  import java.util.Map;
22  
23  
24  class ArgumentValidatorImpl<O> implements ArgumentValidator<O>
25  {
26     private final ValidationErrorBuilder m_validationErrorBuilder;
27     private final List<String> m_validatedUnparsedArguments;
28     private final Map<String, List<String>> m_validatedArguments;
29     private final OptionsSpecification<O> m_specification;
30  
31     public ArgumentValidatorImpl(final OptionsSpecification<O> specification)
32     {
33        m_specification = specification;
34  
35        m_validatedArguments = new LinkedHashMap<String, List<String>>();
36        m_validatedUnparsedArguments = new ArrayList<String>();
37  
38        m_validationErrorBuilder = new ValidationErrorBuilderImpl();
39     }
40  
41     /**
42      * {@inheritDoc}
43      */
44     public ArgumentCollection validateArguments(final ArgumentCollection arguments) throws ArgumentValidationException
45     {
46        m_validatedUnparsedArguments.addAll(arguments.getUnparsed());
47  
48        final Iterator<Argument> argumentsIterator = arguments.iterator();
49        while (argumentsIterator.hasNext())
50        {
51           final Argument argument = argumentsIterator.next();
52           final boolean isLast = !argumentsIterator.hasNext();
53  
54           if(!m_specification.isSpecified(argument.getOptionName()))
55           {
56              m_validationErrorBuilder.unexpectedOption(argument.getOptionName());
57           }
58           else
59           {
60              final OptionSpecification optionSpecification = m_specification.getSpecification(argument.getOptionName());
61              if(optionSpecification.isHelpOption())
62              {
63                 m_validationErrorBuilder.helpRequested(m_specification);
64              }
65              else if(argument.getValues().size() == 0 && optionSpecification.hasValue() && !optionSpecification.isMultiValued())
66              {
67                 m_validationErrorBuilder.missingValue(optionSpecification);
68              }
69              else if(!isLast && argument.getValues().size() > 0 && !optionSpecification.hasValue())
70              {
71                 m_validationErrorBuilder.unexpectedValue(optionSpecification);
72              }
73              else if(!isLast && argument.getValues().size() > 1 && !optionSpecification.isMultiValued())
74              {
75                 m_validationErrorBuilder.unexpectedAdditionalValues(optionSpecification);
76              }
77  
78              if(isLast && hasExcessValues(argument, optionSpecification))
79              {
80                 final List<String> values = new ArrayList<String>();
81                 final List<String> unparsed;
82                 if(optionSpecification.hasValue())
83                 {
84                    values.add(argument.getValues().get(0));
85                    unparsed = new ArrayList<String>(argument.getValues().subList(1, argument.getValues().size()));
86                 }
87                 else
88                 {
89                    unparsed = argument.getValues();
90                 }
91  
92                 m_validatedArguments.put(argument.getOptionName(), values);
93                 m_validatedUnparsedArguments.addAll(0, unparsed);
94              }
95              else
96              {
97                 checkAndAddValues(optionSpecification, argument.getOptionName(), new ArrayList<String>(argument.getValues()));
98              }
99           }
100       }
101 
102       for(final OptionSpecification optionSpecification : m_specification.getMandatoryOptions())
103       {
104          if(!(arguments.containsAny(OptionUtils.getAllNames(optionSpecification))))
105          {
106             m_validationErrorBuilder.missingOption(optionSpecification);
107          }
108       }
109 
110       validateUnparsedOptions();
111 
112       m_validationErrorBuilder.validate();
113 
114       return new ArgumentCollectionImpl(m_validatedArguments, m_validatedUnparsedArguments);
115    }
116 
117    private void validateUnparsedOptions()
118    {
119       if(m_specification.hasUnparsedSpecification())
120       {
121          final OptionSpecification argumentSpecification = m_specification.getUnparsedSpecification();
122 
123          if(!argumentSpecification.isMultiValued() && !argumentSpecification.isOptional() && m_validatedUnparsedArguments.isEmpty())
124          {
125             m_validationErrorBuilder.missingValue(argumentSpecification);
126          }
127          else if (!argumentSpecification.isMultiValued() && m_validatedArguments.size() > 1)
128          {
129             m_validationErrorBuilder.unexpectedValue(argumentSpecification);
130          }
131       }
132    }
133 
134    private boolean hasExcessValues(final Argument entry, final OptionSpecification optionSpecification)
135    {
136       return (!optionSpecification.isMultiValued()
137                 && (entry.getValues().size() > 1 || (entry.getValues().size() > 0 && !optionSpecification.hasValue())));
138    }
139 
140    private void checkAndAddValues(final OptionSpecification optionSpecification, final String option, final ArrayList<String> values)
141    {
142       for (final String value : values)
143       {
144          if(!patternMatches(optionSpecification, value))
145          {
146             m_validationErrorBuilder.patternMismatch(optionSpecification, value);
147          }
148       }
149       m_validatedArguments.put(option, new ArrayList<String>(values));
150    }
151 
152    private boolean patternMatches(final OptionSpecification optionSpecification, final String value)
153    {
154       return value.matches(optionSpecification.getPattern());
155    }
