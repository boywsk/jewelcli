14  package uk.co.flamingpenguin.jewel.cli;
15  
16  import java.util.ArrayList;
17  import java.util.List;
18  
19  import com.lexicalscope.fluentreflection.ReflectedClass;
20  import com.lexicalscope.fluentreflection.ReflectedMethod;
21  
22  class ParsedOptionSpecificationBuilder implements OptionSpecificationBuilder {
23      private final ReflectedMethod m_method;
24      private ReflectedClass<?> m_type;
25      private boolean m_multiValued;
26      private ReflectedMethod optionalityMethod;
27      private final List<String> m_shortNames = new ArrayList<String>();
28      private List<String> m_longName;
29      private String m_description;
30      private String m_pattern;
31      private List<String> m_defaultValue;
32      private boolean m_helpRequest;
33      private boolean defaultToNull;
34  
35      public ParsedOptionSpecificationBuilder(final ReflectedMethod method) {
36          m_method = method;
37      }
38  
39      public void setType(final ReflectedClass<?> type) {
40          m_type = type;
41      }
42  
43      public void setMultiValued(final boolean multiValued) {
44          m_multiValued = multiValued;
45      }
46  
47      @Override public void setOptionalityMethod(final ReflectedMethod optionalityMethod) {
48          this.optionalityMethod = optionalityMethod;
49      }
50  
51      public void setShortNames(final List<String> shortNames) {
52          m_shortNames.addAll(shortNames);
53      }
54  
55      public void setLongName(final List<String> longName) {
56          m_longName = longName;
57      }
58  
59      public void setDescription(final String description) {
60          m_description = description;
61      }
62  
63      public void setPattern(final String pattern) {
64          m_pattern = pattern;
65      }
66  
67      public void setDefaultValue(final List<String> defaultValue) {
68          m_defaultValue = defaultValue;
69      }
70  
71      public void setDefaultToNull(final boolean defaultToNull) {
72          this.defaultToNull = defaultToNull;
73      }
74  
75      public void setHelpRequest(final boolean helpRequest) {
76          m_helpRequest = helpRequest;
77      }
78  
79      public ParsedOptionSpecification createOptionSpecification() {
80          final OptionName optionName =
81                  new OptionName(m_method, m_method.propertyName(), m_longName, m_shortNames, m_description);
82          final OptionType optionType = new OptionType(m_type.classUnderReflection(), m_pattern, m_multiValued);
83          final OptionContext optionContext = new OptionContext(m_defaultValue, m_helpRequest, defaultToNull);
84  
85          return new ParsedOptionSpecificationImpl(optionName,
86                  optionType,
87                  optionContext,
88                  m_method,
89                  optionalityMethod);
90      }
91  }
