14  package uk.co.flamingpenguin.jewel.cli;
15  
16  import java.util.ArrayList;
17  import java.util.List;
18  
19  import com.lexicalscope.fluentreflection.ReflectedMethod;
20  
21  class ParsedOptionSpecificationBuilder extends AbstractOptionSpecificationBuilder implements OptionSpecificationBuilder {
22      private final List<String> m_shortNames = new ArrayList<String>();
23      private List<String> m_longName;
24      private boolean m_helpRequest;
25  
26      public ParsedOptionSpecificationBuilder(final ReflectedMethod method) {
27          super(method);
28      }
29  
30      public void setShortNames(final List<String> shortNames) {
31          m_shortNames.addAll(shortNames);
32      }
33  
34      public void setLongName(final List<String> longName) {
35          m_longName = longName;
36      }
37  
38      public void setHelpRequest(final boolean helpRequest) {
39          m_helpRequest = helpRequest;
40      }
41  
42      public ParsedOptionSpecification createOptionSpecification() {
43          final OptionName optionName =
44                  new OptionName(method, method.propertyName(), m_longName, m_shortNames, description);
45          final OptionType optionType = new OptionType(type.classUnderReflection(), pattern, multiValued);
46          final OptionContext optionContext = new OptionContext(defaultValue, m_helpRequest, defaultToNull);
47  
48          return new ParsedOptionSpecificationImpl(optionName,
49                  optionType,
50                  optionContext,
51                  method,
52                  optionalityMethod);
53      }
54  }
