7   import java.lang.reflect.Method;
8   import java.util.List;

10  import com.lexicalscope.fluentreflection.ReflectedMethod;

12  /**
13   * Specifies an Option
14   * 
15   * @author t.wood
16   */
17  interface OptionSpecification
18  {
19      /**
20       * Each argument to this option must conform to the type returned by this
21       * method
22       * 
23       * @return the type that each argument must conform to
24       */
25      Class<?> getType();
26  
27      /**
28       * Does the option take any arguments?
29       * 
30       * @return True iff the the option takes at least one argument
31       */
32      boolean hasValue();
33  
34      /**
35       * Are multiple arguments allowed?
36       * 
37       * @return True iff the the option takes multiple arguments
38       */
39      boolean isMultiValued();
40  
41      /**
42       * Is the argument optional
43       * 
44       * @return is the argument optional
45       */
46      boolean isOptional();
47  
48      /**
49       * Get all of the short names of this option. Short names are single
50       * characters that will by prefixed by the user with "-".
51       * 
52       * @return the short names of this option
53       */
54      List<String> getShortNames();
55  
56      List<String> getNames();
57  
58      /**
59       * Does this option have a Short Name
60       * 
61       * @return true iff the options has a short name
62       */
63      boolean hasShortName();
64  
65      /**
66       * Get the long name of this option. Long names are multiple characters that
67       * will be prefixed by the user with "--".
68       * 
69       * @return the long name of this option
70       */
71      String getLongName();
72  
73      /**
74       * Get the default values which will be used if the option is not specified
75       * by the user.
76       * 
77       * @return The default values which will be used if the option is not
78       *         present
79       */
80      List<String> getDefaultValue();
81  
82      /**
83       * Is there a default value to use if this option is not present? Options
84       * with a default value are assumed to be optional.
85       * 
86       * @return True iff this option has a default value
87       */
88      boolean hasDefaultValue();
89  
90      /**
91       * Get a description of the option. The description can be specified in the
92       * <code>Option</code> annotation
93       * 
94       * @see uk.co.flamingpenguin.jewel.cli.Option
95       * 
96       * @return a description of the option
97       */
98      String getDescription();
99  
100     /**
101      * Is this option a request for help
102      * 
103      * @return True iff this option is a request for help
104      */
105     boolean isHelpOption();
106 
107     /**
108      * The pattern that values must conform to
109      * 
110      * @return the regular expression that values must conform to
111      */
112     String getPattern();
113 
114     Method getMethod();
115 
116     ReflectedMethod getOptionalityMethod();
